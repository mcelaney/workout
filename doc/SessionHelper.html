def account_link
  if current_user?
    link_to I18n.t('user.show'), user_path(current_user)
  else
    link_to I18n.t('navigation.sign_up'), new_user_path
  end
end

def authentication_link
  if current_user?
    link_to I18n.t('navigation.sign_out'), log_out_path
  else
    link_to I18n.t('navigation.sign_in'), log_in_path
  end
end
